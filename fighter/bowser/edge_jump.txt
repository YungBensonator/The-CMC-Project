{
  switch ( timer )
      {
	case 4: image_index = 1; image_angle = 45-facing*90; shakex = -10+facing*20; break;

	instance_create(x-size,y+size,obj_spark_effect);

	{
		state_type = "jump";
        sound_play(jump_sound);

        vspeed = -6*size;
    }
    if ( timer == 15 )
	intangible = 0;
}
