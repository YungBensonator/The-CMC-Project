if ( state_type = "jump" )
{
  state_type = "air_attack";
  attack_type = "uair";
  timer = 0;
  time = 27;
  sprite_index = uair_sprite;
  image_speed = 0;
  image_index = 1;
}
if ( timer < time ) 
{
  switch ( timer )
  {
    case 3: image_index = 2; break;
    case 5: image_index = 3; break;
    case 7: image_index = 4; break;
    case 9: image_index = 5; break;
    case 11: image_index = 0; break;
  }
  if ( timer = 5 )
  {
    play_attack_voice();
    sound_play (swingsnd1);
    create_hitbox(6,70+facing*40,5.6,2.6,7,4,slap_m,obj_spark_hit1,0.7,1.4,18,17,self.id,6,1);
    hitbox.hit_stun = 8;
  }
  if ( timer = 7 )
  {
    hitbox.hitbox_x = (3-facing*6)*size;
    hitbox.hitbox_y = 40*size;
    hitbox.image_xscale = 1.2*size;
    hitbox.image_yscale = 0.7*size;
  }
  if ( timer = 9 )
  {
	hitbox.sfx = slap_s
    hitbox.bkb = 4.6*size;
    hitbox.kbg = 2.4*size;
    hitbox.hit_stun = 7;
    hitbox.hitbox_x = (-16+facing*32)*size;
    hitbox.hitbox_y = 26*size;
    hitbox.image_xscale = 0.7*size;
    hitbox.image_yscale = 1.2*size;
    hitbox.dir = 165-facing*150;
  }
  if ( air = 0 )
  {
    state_type = "land";
    sprite_index = land_sprite;
    attack_type ="none";
    timer = 0;
    time = 5;
  }
}

if ( timer >= time-1 && state_type != "flinch" && state_type != "air_flinch" )
{
  attack_type = "none";
  sprite_index = jump_sprite;
  image_index = 6;
  image_speed = 0;
  if air > 0
    state_type = "jump";
  else
    state_type = "stand";
  timer = 0;
  time = -1;
}
