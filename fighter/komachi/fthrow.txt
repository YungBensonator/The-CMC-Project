		{
			target.x = x + 2 * size * (1-facing*2);

		{
			sound_play(fthrow_voice);
		}

		{
			//sound_play(attack_sound);
			create_throw_box(1,50+facing*80,1,0,5,1,slash2,obj_spark_slash1,0.7,1.2,20,20,self.id,1,2);

			hitbox.hit_stun = 19;
			hitbox.hitshake_type = 1

}
