if ( time = -1 )
  {
    state_type = "attack";
    attack_type = "bthrow";
    timer = 0;
    time = 31;
    sprite_index = bthrow_sprite;
    image_speed = 0;
    image_index = 0;
    hspeed = 0;
    target.timer_speed = 0;
    target.alarm[0] = 50;
  }

if ( timer < time )
  {
    switch ( timer )
      {
	case 3: image_index = 1; break;
	case 6: image_index = 2; break;
	case 9: image_index = 3; break;
	case 11: image_index = 4; break;
	case 13: image_index = 5; break;
	case 15: image_index = 6; break;
      }

    if ( timer >= 0 && timer <= 2  )
      {
		target.x = x+0*(1+facing*2) * size;
		target.y = y-12*size;
      }
    if ( timer >= 3 && timer <= 5 )
      {
		target.x = x-2*(1+facing*2) * size;
		target.y = y-6*size;
      }
    if ( timer >= 6 && timer <= 8 )
      {
		target.x = x-2*(1+facing*2) * size;
		target.y = y-9*size;
      }
    if ( timer >= 9 && timer <= 11 )
      {
		target.x = x-1*(1+facing*2) * size;
		target.y = y-7*size;
      }
	
    if ( timer = 11 )
      {
		target.x = x - 6*size*(1-facing*2);
		target.y = y - 2*size;
		play_attack_voice();
		create_throw_box(7,150-facing*120,6.2,1.5,1,-5,-1,-1,0.8,0.8,-10,14,self.id,1);
		hitbox.weight_kb = 3
		hitbox.di = 0.5;
		hitbox.hit_stun = 7;
      }
  }

if ( timer >= time-1 && state_type != "flinch" && state_type != "air_flinch" )
  {
    state_type = "stand";
    attack_type = "none";
    timer = 0;
    time = -1;
  }
